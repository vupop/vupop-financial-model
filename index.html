            document.getElementById('loginScreen').style.display = 'flex';

            document.getElementById('mainApp').style.display = 'none';

            document.getElementById('password').value = '';

        }

        

        // Check if user is already logged in

        window.addEventListener('load', function() {

            if (sessionStorage.getItem('vupopLoggedIn') === 'true') {

                document.getElementById('loginScreen').style.display = 'none';

                document.getElementById('mainApp').style.display = 'block';

                calculateProjections();

            }

        });

        

        // User growth assumptions

        const mauTargets = [50000, 150000, 313000, 500000, 750000];

        

        // B2B customer assumptions

        const b2bCustomers = {

            social: [3, 5, 15, 25, 40],

            broadcast: [3, 8, 15, 25, 35],

            broadcastPlus: [1, 3, 5, 8, 10]

        };

        

        const usageSeconds = [2900, 25000, 100000, 250000, 450000];

        

        function formatCurrency(amount) {

            if (amount >= 1000000) {

                return '£' + (amount / 1000000).toFixed(1) + 'M';

            } else if (amount >= 1000) {

                return '£' + (amount / 1000).toFixed(0) + 'K';

            } else {

                return '£' + amount.toLocaleString();

            }

        }

        

        function calculateProjections() {

            // Get input values

            const premiumPrice = parseFloat(document.getElementById('premiumPrice').value);

            const adoptionRate = parseFloat(document.getElementById('adoptionRate').value) / 100;

            const adRevenue = parseFloat(document.getElementById('adRevenue').value);

            const affiliateRate = parseFloat(document.getElementById('affiliateRate').value) / 100;

            const socialTierPrice = parseFloat(document.getElementById('socialTierPrice').value);

            const broadcastTierPrice = parseFloat(document.getElementById('broadcastTierPrice').value);

            const usageFee = parseFloat(document.getElementById('usageFee').value);

            

            // Calculate projections for each year

            const projections = [];

            

            for (let year = 0; year < 5; year++) {

                const mau = mauTargets[year];

                

                // B2C Revenue calculations

                const premiumSubscriptions = mau * adoptionRate * premiumPrice * 12;

                const adRevenueYear = year >= 2 ? mau * adRevenue * 12 : 0; // Ad revenue from Year 3

                const affiliateRevenue = year === 0 ? 18360 : 

                                       year === 1 ? 54120 :

                                       year === 2 ? 881120 :

                                       year === 3 ? 2514000 : 5001000;

                

                const totalB2C = premiumSubscriptions + adRevenueYear + affiliateRevenue;

                

                // B2B Revenue calculations

                const socialRevenue = b2bCustomers.social[year] * socialTierPrice * 12;

                const broadcastRevenue = b2bCustomers.broadcast[year] * broadcastTierPrice * 12;

                const broadcastPlusRevenue = b2bCustomers.broadcastPlus[year] * 4997 * 12; // Fixed price

                const usageRevenue = usageSeconds[year] * usageFee;

                

                const totalB2B = socialRevenue + broadcastRevenue + broadcastPlusRevenue + usageRevenue;

                

                const totalRevenue = totalB2C + totalB2B;

                

                // Cost calculations (simplified)

                const cogs = totalRevenue * (year === 0 ? 0.49 : year === 1 ? 0.40 : year === 2 ? 0.30 : year === 3 ? 0.28 : 0.25);

                const opex = year === 0 ? 950000 : year === 1 ? 1900000 : year === 2 ? 3200000 : year === 3 ? 5200000 : 7500000;

                const grossProfit = totalRevenue - cogs;

                const ebitda = grossProfit - opex;

                const ebitdaMargin = (ebitda / totalRevenue) * 100;

                

                projections.push({

                    mau,

                    b2cRevenue: totalB2C,

                    b2bRevenue: totalB2B,

                    totalRevenue,

                    ebitda,

                    ebitdaMargin

                });

            }

            

            // Update the table

            for (let i = 0; i < 5; i++) {

                const year = i + 1;

                document.getElementById(`mau${year}`).textContent = projections[i].mau.toLocaleString();

                document.getElementById(`b2c${year}`).textContent = formatCurrency(projections[i].b2cRevenue);

                document.getElementById(`b2b${year}`).textContent = formatCurrency(projections[i].b2bRevenue);

                document.getElementById(`total${year}`).textContent = formatCurrency(projections[i].totalRevenue);

                document.getElementById(`ebitda${year}`).textContent = formatCurrency(projections[i].ebitda);

                document.getElementById(`margin${year}`).textContent = Math.round(projections[i].ebitdaMargin) + '%';

            }

            

            // Update key metrics

            const year4Revenue = projections[3].totalRevenue;

            const exitValuation = year4Revenue * 10; // 10x multiple

            const returnMultiple = exitValuation / 4500000; // From £4.5M current valuation

            const year4Ebitda = projections[3].ebitda;

            

            document.getElementById('year4Revenue').textContent = formatCurrency(year4Revenue);

            document.getElementById('exitValuation').textContent = formatCurrency(exitValuation);

            document.getElementById('returnMultiple').textContent = returnMultiple.toFixed(1) + 'x';

            document.getElementById('ebitdaYear4').textContent = formatCurrency(year4Ebitda);

        }

        

        // Add event listeners to inputs

        document.querySelectorAll('input').forEach(input => {

            input.addEventListener('input', calculateProjections);

        });

        

        // Initial calculation

        calculateProjections();

    </script>
